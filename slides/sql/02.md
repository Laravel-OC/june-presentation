The vanilla PHP way

```php
<?php // vanilla php

// First, declare a repo class

class BaseRepo
{
    private $conn;

    public function __construct(PDO $conn)
    {
        $this->conn = $conn;
    }

    public function fetchByQuery($query)
    {
        $stmt = $this->conn->prepare($query);
        $stmt->execute();

        //TODO: error checking
        //TODO: fetch style
        //TODO: normalize results (ints, strings, datetimes, etc.)
        return $stmt->fetch();
    }
}
```
